Differential gene expression analysis: Non-lactating vs Lactating (Fig. 13)

# Convert SCE to Seurat object
rownames(sce) <- make.unique(rownames(sce))

seurat_obj <- as.Seurat(sce)

# Create a LactationStatus column
seurat_obj$Lactation_Status <- ifelse(
  seurat_obj$Epithelial_clusters %in% c("Luminal_HMC1", "Luminal_HMC2"),
  "Lactating",
  ifelse(
    seurat_obj$Epithelial_clusters %in% c("Basal", "Luminal_LP", "Luminal_HR"),
    "Non-Lactating",
    NA
  )
)


# Run differential expression for gene list
markers <- FindMarkers(
  object = seurat_obj,
  ident.1 = "Lactating",
  ident.2 = "Non-Lactating",
  group.by = "Lactation_Status",
  features = genes_bma,
  test.use = "wilcox"
)
# Volcano Plot

# Add gene names as a column
markers$gene <- rownames(markers)

# Define which genes to label (significant & high fold-change)
markers_to_label <- subset(markers, p_val_adj < 0.01 & abs(avg_log2FC) > 1)

# Volcano plot with repelled labels
volcano <- ggplot(markers, aes(x = avg_log2FC, y = -log10(p_val_adj))) +
  geom_point(alpha = 0.6) +
  geom_vline(xintercept = c(-1, 1), linetype = "dashed", color = "grey") +
  geom_hline(yintercept = -log10(0.05), linetype = "dashed", color = "red") +
  geom_text_repel(
    data = markers_to_label,
    aes(label = gene),
    size = 3,
    max.overlaps = 50,
    box.padding = 0.5,
    point.padding = 0.2,
    force = 1,
    segment.color = "grey50"
  ) +
  theme_minimal() +
  labs(
    title = "Lactating vs Non-Lactating Cells",
    x = "Log2 Fold Change",
    y = "-log10 Adjusted P-value"
  )

ggsave("~/Lactation Lab/MPhil_project/single_cell_RNAseq_human_milk/outputs/volcano_plot_Thesis_DEGs.png", plot = volcano, width = 8, height = 6, dpi = 300)
